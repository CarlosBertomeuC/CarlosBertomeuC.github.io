2. Programación Orientada a Objetos (POO)

Conceptos Básicos


public class Persona
{
    public string Nombre { get; set; }
    public int Edad { get; set; }

    public void Presentarse()
    {
        Console.WriteLine("Hola, mi nombre es " + Nombre + " y tengo " + Edad + " años.");
    }
}

// Creación y uso de un objeto
Persona persona = new Persona();
persona.Nombre = "Juan";
persona.Edad = 30;
persona.Presentarse();

Principios de POO

Encapsulamiento


public class CuentaBancaria
{
    private double saldo;

    public double ObtenerSaldo()
    {
        return saldo;
    }

    public void Depositar(double cantidad)
    {
        if (cantidad > 0)
        {
            saldo += cantidad;
        }
    }

    public void Retirar(double cantidad)
    {
        if (cantidad > 0 && saldo >= cantidad)
        {
            saldo -= cantidad;
        }
    }
}

Herencia


public class Animal
{
    public string Nombre { get; set; }

    public void Comer()
    {
        Console.WriteLine(Nombre + " está comiendo.");
    }
}

public class Perro : Animal
{
    public void Ladrar()
    {
        Console.WriteLine(Nombre + " está ladrando.");
    }
}

// Uso de la herencia
Perro perro = new Perro();
perro.Nombre = "Rex";
perro.Comer();
perro.Ladrar();

Polimorfismo


public class Vehiculo
{
    public virtual void Conducir()
    {
        Console.WriteLine("Conduciendo un vehículo.");
    }
}

public class Coche : Vehiculo
{
    public override void Conducir()
    {
        Console.WriteLine("Conduciendo un coche.");
    }
}

public class Moto : Vehiculo
{
    public override void Conducir()
    {
        Console.WriteLine("Conduciendo una moto.");
    }
}

// Uso del polimorfismo
Vehiculo miVehiculo1 = new Coche();
Vehiculo miVehiculo2 = new Moto();
miVehiculo1.Conducir();
miVehiculo2.Conducir();

Abstracción


public abstract class Figura
{
    public abstract double CalcularArea();
}

public class Circulo : Figura
{
    public double Radio { get; set; }

    public override double CalcularArea()
    {
        return Math.PI * Radio * Radio;
    }
}

public class Rectangulo : Figura
{
    public double Ancho { get; set; }
    public double Alto { get; set; }

    public override double CalcularArea()
    {
        return Ancho * Alto;
    }
}

// Uso de la abstracción
Figura miCirculo = new Circulo { Radio = 5 };
Figura miRectangulo = new Rectangulo { Ancho = 4, Alto = 6 };
Console.WriteLine("Área del círculo: " + miCirculo.CalcularArea());
Console.WriteLine("Área del rectángulo: " + miRectangulo.CalcularArea());
